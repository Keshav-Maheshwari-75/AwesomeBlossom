        // Collect data from admin form
        function collectFormData() {
            // Helper function to parse JSON or return empty array
            const parseJson = (str, defaultValue = []) => {
                try {
                    return JSON.parse(str);
                } catch {
                    return defaultValue;
                }
            };
            
            // Helper function to split lines and trim
            const splitLines = (str) => {
                return str.split('\n').map(line => line.trim()).filter(line => line);
            };
            
            return {
                hero: {
                    title: document.getElementById('hero-title').value,
                    tagline: document.getElementById('hero-tagline').value,
                    bgImage: document.getElementById('hero-bg').value,
                    date: document.getElementById('event-date').value,
                    location: document.getElementById('event-location').value,
                    ctaText: document.getElementById('cta-text').value
                },
                eventDetails: {
                    description: document.getElementById('event-description').value,
                    schedule: parseJson(document.getElementById('event-schedule').value),
                    audience: document.getElementById('target-audience').value,
                    dressCode: document.getElementById('dress-code').value
                },
                venue: {
                    name: document.getElementById('venue-name').value,
                    address: document.getElementById('venue-address').value,
                    transport: document.getElementById('venue-transport').value,
                    parking: document.getElementById('venue-parking').value,
                    map: document.getElementById('venue-map').value
                },
                registration: {
                    tickets: parseJson(document.getElementById('ticket-options').value)
                },
                speakers: parseJson(document.getElementById('speakers-list').value),
                gallery: splitLines(document.getElementById('gallery-images').value),
                contact: {
                    phone: document.getElementById('contact-phone').value,
                    email: document.getElementById('contact-email').value,
                    whatsapp: document.getElementById('contact-whatsapp').value,
                    social: parseJson(document.getElementById('social-links').value)
                },
                organizers: {
                    name: document.getElementById('organizer-name').value,

                faq: parseJson(document.getElementById('faq-items').value)
            };
        }
        
        // Countdown Timer
        function updateCountdown() {
            const eventData = JSON.parse(localStorage.getItem('eventData')) || defaultData;
            const eventDate = new Date(eventData.hero.date).getTime();
            const now = new Date().getTime();
            const distance = eventDate - now;
            
            const days = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);
            
            document.getElementById('days').innerHTML = days.toString().padStart(2, '0');
            document.getElementById('hours').innerHTML = hours.toString().padStart(2, '0');
            document.getElementById('minutes').innerHTML = minutes.toString().padStart(2, '0');
            document.getElementById('seconds').innerHTML = seconds.toString().padStart(2, '0');
            
            if (distance < 0) {
                document.getElementById('days').innerHTML = '00';
                document.getElementById('hours').innerHTML = '00';
                document.getElementById('minutes').innerHTML = '00';
                document.getElementById('seconds').innerHTML = '00';
            }
        }
        
        // Initialize Accordion
        function initAccordion() {
            const accordionHeaders = document.querySelectorAll('.accordion-header');
            
            accordionHeaders.forEach(header => {
                header.addEventListener('click', () => {
                    const accordionItem = header.parentElement;
                    const accordionContent = header.nextElementSibling;
                    
                    if (header.classList.contains('active')) {
                        header.classList.remove('active');
                        accordionContent.classList.remove('show');
                    } else {
                        accordionHeaders.forEach(item => {
                            item.classList.remove('active');
                            item.nextElementSibling.classList.remove('show');
                        });
                        
                        header.classList.add('active');
                        accordionContent.classList.add('show');
                    }
                });
            });
        }
        
        // Helper function to strip HTML tags
        function stripHtml(html) {
            const tmp = document.createElement('div');
            tmp.innerHTML = html;
            return tmp.textContent || tmp.innerText || '';
        }
